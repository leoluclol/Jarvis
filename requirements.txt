# =====================================================================
# TARGET: Raspberry Pi 2 - Raspberry Pi OS 12 "Bookworm" 32-bit (armv7l)
#         Python 3.11 (the system Python on Bookworm)
#
# Every pin below is available as a PREBUILT armv7l wheel on piwheels,
# so a Pi 2 compiles nothing. Do not bump these casually: numpy >= 2.x and
# scipy >= 1.16 have armv7l wheels only for newer Python, and building
# either from source on a Pi 2 takes hours.
#
# Python 3.9 (Bullseye) also works - the same pins have cp39 wheels.
#
# ---------------------------------------------------------------------
# INSTALL (two steps - the second one MUST use --no-deps, see below)
#
#   sudo apt install -y portaudio19-dev libopenblas-dev
#   pip install -r requirements.txt
#   pip install --no-deps -r requirements-raven.txt
#
# On the Pi, point pip at piwheels first so the wheels are actually found:
#   printf '[global]\nextra-index-url=https://www.piwheels.org/simple\n' \
#     | sudo tee /etc/pip.conf
#
# Then record your wake word (Raven ships no pretrained models):
#   python jarvis.py --record
# ---------------------------------------------------------------------
#
# WHY --no-deps IS MANDATORY, NOT A WORKAROUND:
# rhasspy-wake-raven hardcodes `scipy==1.5.1` in its metadata. scipy 1.5.1
# only ever published cp36/cp37/cp38 wheels, so on any Python >= 3.9 pip
# falls back to building it from source. That build pulls an ancient numpy
# whose distutils shim is broken on modern setuptools, which is the
# `NameError: name 'CCompiler' is not defined` you hit.
# The pin is stale, not a real constraint: Raven only calls
# scipy.io.wavfile.read. Installing it with --no-deps against scipy 1.11.4
# is tested and works.
# =====================================================================

# --- Numeric core (MFCC + DTW math) ---
numpy==1.26.4
scipy==1.11.4

# --- Audio capture and VAD ---
PyAudio==0.2.14
webrtcvad==2.0.10

# RUNTIME dependency, not a build tool: webrtcvad 2.0.10 does `import
# pkg_resources` at the top of webrtcvad.py, and pkg_resources ships inside
# setuptools. setuptools 83.0.0 deleted pkg_resources, so anything >= 83
# makes `import webrtcvad` raise ModuleNotFoundError. Modern Python does not
# preinstall setuptools in a venv, so it must be pinned explicitly here.
setuptools==75.8.0

# --- OpenAI client and utilities ---
openai==1.60.2
python-dotenv==1.0.1

# --- Resource logging ---
psutil==5.9.3
